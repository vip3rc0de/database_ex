1st View BOSS_MIS
---------------------------------
select
    PROSWPIKO.ONOMA_PR , PROSWPIKO.EPONYMO_PR, PROSWPIKO.MISTHOS
from
     PROSWPIKO
where
     PROSWPIKO.TITLOS = 'Boss'
---------------------------------
